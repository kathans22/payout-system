// Relationships
Client -> API_Gateway : HTTP Request (Rupees to Paise boundary conversion)
API_Gateway -> SaleService : POST /sales\nPOST /sales/:id/reconcile
API_Gateway -> AdvancePayoutJobService : POST /jobs/advance-payout
API_Gateway -> LedgerService : GET /users/:id/balance\nGET /users/:id/ledger
API_Gateway -> PayoutService : POST /users/:id/withdraw\nPOST /payouts/:id/fail

SaleService -> SaleCollection : Write/Update Sale
SaleService -> LedgerCollection : Write credit/debit entries (Transaction session)

AdvancePayoutJobService -> SaleCollection : Query pending sales
AdvancePayoutJobService -> LedgerCollection : Write 10% ADVANCE (idempotent, catch 11000)

LedgerService -> LedgerCollection : Aggregates sum of amountPaise

PayoutService -> PayoutCollection : Create Payout / update status
PayoutService -> LedgerCollection : Debit/Reversal entry (Transaction session)
PayoutService -> LedgerService : Verifies current derived balance
PayoutService -> WithdrawalService : Enforces 24-hour limit check

WithdrawalService -> LedgerCollection : Find most recent WITHDRAWAL entry

UserCollection -> MongoDB
BrandCollection -> MongoDB
SaleCollection -> MongoDB
LedgerCollection -> MongoDB
PayoutCollection -> MongoDB
